# createPikas

You can create yout custom configuration for pikas by using the createPikas function.

If you want to create a custom configuration, add this file to your project:

```ts
import type { CSS } from '@pikas-ui/styles'
import {
  createPikas,
  loadBorderStyles,
  loadBorderWidths,
  loadColors,
  loadFontSizes,
  loadFontWeights,
  loadLetterSpacings,
  loadLineHeights,
  loadRadii,
  loadShadows,
  loadSizes,
  loadSpaces,
  loadTransitions,
  loadZIndices,
  loadFonts,
  loadMedias,
  loadUtils,
} from '@pikas-ui/styles'

export const colors = loadColors({
  // your colors
})
export type Colors = typeof colors
export type Color = keyof Colors

export const fontSizes = loadFontSizes({
  // your font sizes
})
export type FontSizes = typeof fontSizes
export type FontSize = keyof FontSizes

export const fontWeights = loadFontWeights({
  // your font weights
})
export type FontWeights = typeof fontWeights
export type FontWeight = keyof FontWeights

export const letterSpacings = loadLetterSpacings({
  // your letter spacings
})
export type LetterSpacings = typeof letterSpacings
export type LetterSpacing = keyof LetterSpacings

export const lineHeights = loadLineHeights({
  // your line heights
})
export type LineHeights = typeof lineHeights
export type LineHeight = keyof LineHeights

export const sizes = loadSizes({
  // your sizes
})
export type Sizes = typeof sizes
export type Size = keyof Sizes

export const spaces = loadSpaces({
  // your spaces
})
export type Spaces = typeof spaces
export type Space = keyof Spaces

export const borderStyles = loadBorderStyles({
  // your border styles
})
export type BorderStyles = typeof borderStyles
export type BorderStyle = keyof BorderStyles

export const borderWidths = loadBorderWidths({
  // your border widths
})
export type BorderWidths = typeof borderWidths
export type BorderWidth = keyof BorderWidths

export const radii = loadRadii({
  // your radii
})
export type Radii = typeof radii
export type Radius = keyof Radii

export const shadows = loadShadows({
  // your shadows
})
export type Shadows = typeof shadows
export type Shadow = keyof Shadows

export const transitions = loadTransitions({
  // your transitions
})
export type Transitions = typeof transitions
export type Transition = keyof Transitions

export const zIndices = loadZIndices({
  // your z indices
})
export type ZIndices = typeof zIndices
export type ZIndex = keyof ZIndices

export const fonts = loadFonts({
  // your fonts
})
export type Fonts = typeof fonts
export type Font = keyof Fonts

export const medias = loadMedias({
  // your medias
})
export type Medias = typeof medias
export type Media = keyof Medias

export const utils = loadUtils({
  // your utils
})

export const pikasConfig = createPikas({
  theme: {
    colors: colors,
    fonts: fonts,
    shadows: shadows,
    fontSizes: fontSizes,
    space: spaces,
    sizes: sizes,
    fontWeights: fontWeights,
    lineHeights: lineHeights,
    letterSpacings: letterSpacings,
    borderWidths: borderWidths,
    borderStyles: borderStyles,
    radii: radii,
    zIndices: zIndices,
    transitions: transitions,
  },
  media: medias,
  utils: utils,
})

export const {
  styled,
  css,
  globalCss,
  keyframes,
  getCssText,
  config,
  createTheme,
  prefix,
  reset,
  theme,
} = pikasConfig

export type CustomPikasCSS = CSS<typeof config>

export const customGlobalCss = globalCss({
  'html, body, #__next': {
    fontFamily: '$roboto',
    scrollBehavior: 'smooth',
    color: '$black',
  },
  '*': {
    margin: 0,
    padding: 0,
    boxSizing: 'border-box',
  },
})

export type CustomPikasConfig = GlobalPikasConfig & {
  color: Color
  fontSize: FontSize
  fontWeight: FontWeight
  letterSpacing: LetterSpacing
  lineHeight: LineHeight
  size: Size
  space: Space
  borderStyle: BorderStyle
  borderWidth: BorderWidth
  radius: Radius
  shadow: Shadow
  transition: Transition
  zIndex: ZIndex
  font: Font
  media: Media
}
```

Now you can use your custom configuration in your project.

To apply the custom configuration to your pikas components, you need to pass the config type to the component.

## Example

```tsx
import { Button } from '@pikas-ui/button'
import { CustomPikasConfig } from 'YOUR_CUSTOM_CONFIG_FILE'

export const Example: React.FC = () => {
  return (
    <Button<CustomPikasConfig> onClick={() => console.log('click')}>
      Hello World
    </Button>
  )
}
```
