### Installation

With npm:

```
npm install @pikas-utils/keyboard
```

With yarn:

```
yarn add @pikas-utils/keyboard
```

With pnpm:

```
pnpm add @pikas-utils/keyboard
```

---

## Usage

### useKeyPress

<UseKeyPressExample />

```tsx
import { useKeyPress } from '@pikas-utils/keyboard';

const Example: React.FC = () => {
  const isPressed = useKeyPress('Enter');

  return <div>{isPressed ? 'Enter is pressed' : 'Enter is not pressed'}</div>;
};
```
